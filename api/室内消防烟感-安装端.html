# 1 安装上报模块

# 1.1 扫描设备二维码查询设备

  • 请求URL

/install/codeByImeiData

  • 请求方式

POST

  • 请求参数
请求参数 必选 参数类型 说明
imei true String imei码
hardwareType true Integer 传感器类型 0消火防1烟感器 2消防水压
  • 返回字段
返回字段 字段类型 说明
hardwareName String 传感器名称
hardwareId Integer 传感器设备ID
imei String imei码
deviceName String 消防栓名称

// 正确示例
{
    "content": "",
    "code": "0000",
    "result": {
        "map": {

            "hardwareId": "519665839",
            "imei": "869976036262286",
            "deviceName": "ZXH_XFM_I_QH0014"

    },
    "url": null
}
// 错误示例
{
    "content": "imei码与传感器参数类型不匹配,传感器类型参数传递错误",

# 1.4 扫码后提交表单信息

  • 请求URL

/install/insertCodeInfo

  • 请求方式

POST

  • 请求参数
请求参数 必选 参数类型 说明
deviceName true String 设备名称
hardwareType true Integer 传感器类型 0消火防1烟感器 2消防水压
fireKind true/false Integer 消防栓类型 0室内, 1室外
当hardwareType=0时，fireKind必填
hardwareId true String 硬件传感器ID
hardwareName true String 硬件传感器名称
imei true String imei码
address true String 详细地址
longitude true String 经度
latitude true String 纬度
areaType true/false Integer 场所类型 1社区类，2非社区类
当hardwareType=2时 ，areaType为null
areaType的值由所属场所接口决定
mainId true/false Integer 当areaType=2非社区时，mainId为监管区域ID
当areaType=1社区且hardwareType=0时,mainId为楼层ID
当areaType=1社区且hardwareType=1时,mainId为门户ID
当hardwareType=2时，mainId为null
regulatorsId true/false Integer 监管单位ID
当hardwareType=0时且fireKind=1时，必填
当hardwareType=2时，必填
province true Integer 省编号
city true Integer 市编号
region true Integer 区编号（对应我的模块初始化区县的id字段）
  • 返回字段
返回字段 字段类型 说明
content String 错误提示
code String 0000正确 0001错误
  • 返回示例
//成功提交二维码
{
    "content": "",
    "code": "0000",
    "result": {},
    "url": null
}

//二维码是否使用
{
    "content": "二维码已使用，请勿重复扫码!",
    "code": "0001",
    "result": null,
    "url": ""
}
//错误信息
{
    "content": "缺少必填参数,请检查后重试!",
    "code": "0001",
    "result": null,
    "url": ""
}
